01. SQL Statement:      SELECT DISTINCT
      TO_CHAR(r.rcrtmp1_pidm,'9999999') AS pidm,
      ro.rotiden_stat_ind              AS iden_stat,
      r.rcrtmp1_last_name              AS LastName,
      r.rcrtmp1_first_name             AS FirstName,
      r.rcrtmp1_addr                   AS Address,
      r.rcrtmp1_city                   AS City
 FROM rcrtmp1 r
  JOIN rotiden ro
    ON ro.rotiden_aidy_code = r.rcrtmp1_aidy_code
   AND ro.rotiden_pidm = r.rcrtmp1_pidm
  JOIN rcrtmp4 r4
    ON r4.rcrtmp4_aidy_code = r.rcrtmp1_aidy_code
   AND r4.rcrtmp4_pidm      = r.rcrtmp1_pidm
   AND r4.rcrtmp4_infc_code = r.rcrtmp1_infc_code
   AND r4.rcrtmp4_seq_no    = r.rcrtmp1_seq_no
  LEFT JOIN rcrtmp5 r5
    ON r5.rcrtmp5_aidy_code = r.rcrtmp1_aidy_code
   AND r5.rcrtmp5_pidm      = r.rcrtmp1_pidm
   AND r5.rcrtmp5_infc_code = r.rcrtmp1_infc_code
   AND r5.rcrtmp5_seq_no    = r.rcrtmp1_seq_no
  WHERE r.rcrtmp1_aidy_code = '2324'
    AND r.rcrtmp1_infc_code IN ('EDE','CSS')
